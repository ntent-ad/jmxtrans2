23  package org.jmxtrans.core.output.support;
24  
25  import java.io.IOException;
26  import java.net.MalformedURLException;
27  import java.net.SocketTimeoutException;
28  import java.net.URL;
29  
30  import org.jmxtrans.core.results.QueryResult;
31  import org.jmxtrans.utils.appinfo.AppInfo;
32  import org.jmxtrans.utils.mockito.MockitoTestNGListener;
33  
34  import com.github.tomakehurst.wiremock.WireMockServer;
35  import org.mockito.Mock;
36  import org.testng.annotations.AfterClass;
37  import org.testng.annotations.BeforeClass;
38  import org.testng.annotations.BeforeMethod;
39  import org.testng.annotations.Listeners;
40  import org.testng.annotations.Test;
41  
42  import static java.util.concurrent.TimeUnit.MILLISECONDS;
43  
44  import static com.github.tomakehurst.wiremock.client.WireMock.aResponse;
45  import static com.github.tomakehurst.wiremock.client.WireMock.configureFor;
46  import static com.github.tomakehurst.wiremock.client.WireMock.equalTo;
47  import static com.github.tomakehurst.wiremock.client.WireMock.matching;
48  import static com.github.tomakehurst.wiremock.client.WireMock.post;
49  import static com.github.tomakehurst.wiremock.client.WireMock.postRequestedFor;
50  import static com.github.tomakehurst.wiremock.client.WireMock.resetToDefault;
51  import static com.github.tomakehurst.wiremock.client.WireMock.stubFor;
52  import static com.github.tomakehurst.wiremock.client.WireMock.urlEqualTo;
53  import static com.github.tomakehurst.wiremock.client.WireMock.verify;
54  import static org.assertj.core.api.Assertions.assertThat;
55  
56  @Listeners(MockitoTestNGListener.class)
57  public class HttpOutputWriterTest {
58  
59      private WireMockServer wireMockServer;
60  
61      private AppInfo<AppInfo> appInfo;
62      @Mock private OutputStreamBasedOutputWriter target;
63      @Mock private QueryResult result;
64  
65      @BeforeMethod
66      public void initializeAppInfo() throws IOException {
67          appInfo = AppInfo.load(AppInfo.class);
68      }
69  
70      @BeforeClass
71      public void startHttpServer() {
72          wireMockServer = new WireMockServer();
73          wireMockServer.start();
74          configureFor("localhost", wireMockServer.port());
75      }
76      
77      @BeforeMethod
78      public void resetServer() {
79          resetToDefault();
80      }
81      
82      @Test(
83              expectedExceptions = IllegalArgumentException.class,
84              expectedExceptionsMessageRegExp = "URL needs to be HTTP or HTTPS \\[ftp://ftp.test.net\\]")
85      public void ftpIsNotSupported() throws MalformedURLException {
86          HttpOutputWriter.builder(new URL("ftp://ftp.test.net"), appInfo, target)
87                  .build();
88      }
89  
90      @Test
91      public void canWriteSimplePostMessage() throws IOException {
92          stubFor(post(urlEqualTo("/"))
93                  .willReturn(aResponse()
94                          .withStatus(200)));
95  
96          HttpOutputWriter<DummyStreamWriter> outputWriter = HttpOutputWriter
97                  .builder(new URL("http://localhost:" + wireMockServer.port()), appInfo, new DummyStreamWriter("hello world"))
98                  .build();
99  
100         int count = outputWriter.write(result);
101         assertThat(count).isEqualTo(1);
102 
103         verify(postRequestedFor(urlEqualTo("/"))
104                 .withRequestBody(equalTo("hello world")));
105     }
106 
107     @Test
108     public void canSetContentType() throws IOException {
109         stubFor(post(urlEqualTo("/"))
110                 .withHeader("Content-Type", matching("text/plain"))
111                 .willReturn(aResponse()
112                         .withStatus(200)));
113 
114         HttpOutputWriter<DummyStreamWriter> outputWriter = HttpOutputWriter
115                 .builder(new URL("http://localhost:" + wireMockServer.port()), appInfo, new DummyStreamWriter("hello world"))
116                 .withContentType("text/plain")
117                 .build();
118 
119         int count = outputWriter.write(result);
120         assertThat(count).isEqualTo(1);
121 
122         verify(postRequestedFor(urlEqualTo("/"))
123                 .withHeader("Content-Type", matching("text/plain"))
124                 .withRequestBody(equalTo("hello world")));
125     }
126 
127     @Test
128     public void canAuthenticate() throws IOException {
129         stubFor(post(urlEqualTo("/"))
130                 .willReturn(aResponse()
131                         .withStatus(200)));
132 
133         HttpOutputWriter<DummyStreamWriter> outputWriter = HttpOutputWriter
134                 .builder(new URL("http://localhost:" + wireMockServer.port()), appInfo, new DummyStreamWriter("hello world"))
135                 .withAuthentication("username", "password")
136                 .build();
137 
138         int count = outputWriter.write(result);
139         assertThat(count).isEqualTo(1);
140 
141         verify(postRequestedFor(urlEqualTo("/"))
142                 .withHeader("Authorization", matching("Basic dXNlcm5hbWU6cGFzc3dvcmQ="))
143                 .withRequestBody(equalTo("hello world")));
144     }
145 
146     @Test
147     public void noAuthenticationSentIfAuthenticationNotSet() throws IOException {
148         stubFor(post(urlEqualTo("/"))
149                 .willReturn(aResponse()
150                         .withStatus(200)));
151 
152         HttpOutputWriter<DummyStreamWriter> outputWriter = HttpOutputWriter
153                 .builder(new URL("http://localhost:" + wireMockServer.port()), appInfo, new DummyStreamWriter("hello world"))
154                 .build();
155 
156         int count = outputWriter.write(result);
157         assertThat(count).isEqualTo(1);
158 
159         verify(postRequestedFor(urlEqualTo("/"))
160                 .withoutHeader("Authorization")
161                 .withRequestBody(equalTo("hello world")));
162     }
163 
164     @Test(expectedExceptions = IOException.class, expectedExceptionsMessageRegExp = ".*not OK.*404.*")
165     public void exceptionThrownIfResponseIsNotOk() throws IOException {
166         stubFor(post(urlEqualTo("/"))
167                 .willReturn(aResponse()
168                         .withBody("world hello")
169                         .withStatus(404)));
170 
171         HttpOutputWriter<DummyStreamWriter> outputWriter = HttpOutputWriter
172                 .builder(new URL("http://localhost:" + wireMockServer.port()), appInfo, new DummyStreamWriter("hello world"))
173                 .build();
174 
175         outputWriter.write(result);
176     }
177 
178     @Test(expectedExceptions = SocketTimeoutException.class)
179     public void timeoutIsRespected() throws IOException {
180         stubFor(post(urlEqualTo("/"))
181                 .willReturn(aResponse()
182                         .withFixedDelay(100)
183                         .withStatus(200)));
184 
185         HttpOutputWriter<DummyStreamWriter> outputWriter = HttpOutputWriter
186                 .builder(new URL("http://localhost:" + wireMockServer.port()), appInfo, new DummyStreamWriter("hello world"))
187                 .withTimeout(10, MILLISECONDS)
188                 .build();
189 
190         outputWriter.write(result);
191     }
192     
193     @AfterClass
194     public void stopHttpServer() {
195         wireMockServer.stop();
196     }
197     
198 }

45  public class QueryTest {
46      static MBeanServer mbeanServer = ManagementFactory.getPlatformMBeanServer();
47      static ObjectName mockObjectName;
48      static Mock mock = new Mock("PS Eden Space", 87359488L);
49      private DummyResultNameStrategy resultNameStrategy = new DummyResultNameStrategy();
50  
51      @BeforeClass
52      public static void beforeClass() throws Exception {
53          mockObjectName = new ObjectName("test:type=Mock,name=mock");
54          mbeanServer.registerMBean(mock, mockObjectName);
55      }
56  
57      @AfterClass
58      public static void afterClass() throws Exception {
59          mbeanServer.unregisterMBean(mockObjectName);
60      }
61  
62      @Test
63      public void basic_attribute_return_simple_result() throws Exception {
64          Query query = new Query("test:type=Mock,name=mock", "CollectionUsageThreshold", resultNameStrategy);
65          Queue<QueryResult> results = new ArrayBlockingQueue<QueryResult>(1);
66  
67          query.collectAndExport(mbeanServer, results);
68  
69          assertThat(results).hasSize(1);
70          QueryResult queryResult = results.poll();
71          assertThat(queryResult).isNotNull();
72          assertThat(queryResult.getName()).isEqualTo("CollectionUsageThreshold");
73          assertThat(queryResult.getValue()).isInstanceOf(Number.class);
74      }
75  
76      @Test
77      public void result_name_are_processed() {
78          Query query = new Query("test:type=Mock,name=mock", "CollectionUsageThreshold", resultNameStrategy);
79          Queue<QueryResult> results = new ArrayBlockingQueue<QueryResult>(1);
80  
81          query.collectAndExport(mbeanServer, results);
82  
83          assertThat(resultNameStrategy.hasBeenCalled()).isTrue();
84      }

100     @Test
101     public void non_indexed_list_attribute_return_simple_result() throws Exception {
102         Query query = new Query("test:type=Mock,name=mock", "IntegerList", resultNameStrategy);
103         Queue<QueryResult> results = new ArrayBlockingQueue<QueryResult>(mock.getIntegerList().size());
104 
105         query.collectAndExport(mbeanServer, results);
106 
107         for (int i = 0; i < mock.getIntegerList().size(); i++) {
108             final String name = "IntegerList_" + i;
109             Optional<QueryResult> result = FluentIterable.from(results).firstMatch(new Predicate<QueryResult>() {
110                 @Override
111                 public boolean apply(@Nullable QueryResult input) {
112                     return input != null && name.equals(input.getName());
113                 }
114             });
115 
116             org.assertj.guava.api.Assertions.assertThat(result).isPresent();
117             assertThat(result.get().getValue()).isInstanceOf(Number.class);
118         }
119     }

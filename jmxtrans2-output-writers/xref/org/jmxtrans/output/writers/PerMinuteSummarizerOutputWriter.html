23  package org.jmxtrans.output.writers;
24  
25  import org.jmxtrans.core.output.AbstractOutputWriter;
26  import org.jmxtrans.core.output.OutputWriter;
27  import org.jmxtrans.log.Logger;
28  import org.jmxtrans.log.LoggerFactory;
29  import org.jmxtrans.output.writers.utils.EvictingQueue;
30  import org.jmxtrans.core.results.QueryResult;
31  import org.jmxtrans.core.results.QueryResultComparator;
32  
33  import javax.annotation.Nonnull;
34  import javax.annotation.Nullable;
35  import javax.annotation.concurrent.NotThreadSafe;
36  import java.io.IOException;
37  import java.math.BigDecimal;
38  import java.math.RoundingMode;
39  import java.util.HashMap;
40  import java.util.Map;
41  import java.util.Queue;
42  import java.util.concurrent.TimeUnit;
43  
44  import static java.util.concurrent.TimeUnit.MILLISECONDS;
45  
46  /**
47   * @author <a href="mailto:cleclerc@cloudbees.com">Cyrille Le Clerc</a>
48   */
49  @NotThreadSafe
50  public class PerMinuteSummarizerOutputWriter extends AbstractOutputWriter implements OutputWriter {
51  
52      private final Logger logger = LoggerFactory.getLogger(getClass().getName());
53  
54      protected final OutputWriter delegate;
55      protected final Map<String, Queue<QueryResult>> previousQueryResultsByMetricName = new HashMap<String, Queue<QueryResult>>();
56  
57      public PerMinuteSummarizerOutputWriter(@Nonnull OutputWriter delegate) {
58          this.delegate = delegate;
59      }
60  
61      @Override
62      public void write(@Nonnull QueryResult result) throws IOException {
63  
64          if ("counter".equals(result.getType())) {
65  
66              QueryResult previousResult = getPreviousQueryResult(result);
67  
68              storeQueryResult(result);
69              QueryResult newCurrentResult = perMinute(result, previousResult);
70              logger.debug("Metric " + result.getName() + " is a counter " +
71                      "current=" + result + ", " +
72                      "previous=" + previousResult + ", " +
73                      "newCurrent.value=" + newCurrentResult.getValue());
74  
75              delegate.write(newCurrentResult);
76          } else {
77              logger.debug("Metric " + result.getName() + " is a NOT a counter");
78              delegate.write(result);
79          }
80      }
81  
82      protected void storeQueryResult(@Nullable QueryResult currentResult) {
83          if (currentResult == null)
84              return;
85  
86          Queue<QueryResult> queue = previousQueryResultsByMetricName.get(currentResult.getName());
87          if (queue == null) {
88              queue = new EvictingQueue<>(3);
89              previousQueryResultsByMetricName.put(currentResult.getName(), queue);
90          }
91          queue.add(currentResult);
92      }
93  
94      @Nullable
95      protected QueryResult getPreviousQueryResult(@Nonnull QueryResult currentResult) {
96          Queue<QueryResult> queue = previousQueryResultsByMetricName.get(currentResult.getName());
97          if (queue == null) {
98              return null;
99          }
100 
101         final long targetTimeInMillis = currentResult.getEpoch(MILLISECONDS) - TimeUnit.MILLISECONDS.convert(60, TimeUnit.SECONDS);
102         long closestDistanceToTarget = Long.MAX_VALUE;
103         QueryResult closestQueryResultToTarget = null;
104         for (QueryResult queryResult : queue) {
105             if (new QueryResultComparator().compare(queryResult, currentResult) > 0) {
106                 // skip older result that is greater than current value
107                 // ever increasing counter must be increasing
108             } else {
109                 long distanceToTarget = Math.abs(queryResult.getEpoch(MILLISECONDS) - targetTimeInMillis);
110                 if (distanceToTarget < closestDistanceToTarget) {
111                     closestQueryResultToTarget = queryResult;
112                     closestDistanceToTarget = distanceToTarget;
113                 }
114             }
115         }
116 
117         return closestQueryResultToTarget;
118     }
119 
120     @Nonnull
121     public QueryResult perMinute(@Nonnull QueryResult currentResult, @Nullable QueryResult previousResult) {
122 
123         if (!(currentResult.getValue() instanceof Number)) {
124             logger.info("Current value is not a number, cannot calculate derivative " + currentResult);
125 
126             return currentResult;
127         }
128 
129         if (previousResult == null) {
130             logger.debug("No previous value found for metric '" + currentResult.getName() + "'");
131 
132             return new QueryResult(currentResult.getName(), "gauge", currentResult.getValue(), currentResult.getEpoch(MILLISECONDS));
133         }
134         if (!(previousResult.getValue() instanceof Number)) {
135             logger.info("previous value is not a number, cannot calculate derivative " + previousResult);
136 
137             return currentResult;
138         }
139 
140         BigDecimal durationInMillis = new BigDecimal(currentResult.getEpoch(MILLISECONDS) - previousResult.getEpoch(MILLISECONDS));
141 
142         Number currentValue = (Number) currentResult.getValue();
143         Number previousValue = (Number) previousResult.getValue();
144 
145         if (((Comparable) currentValue).compareTo(previousValue) < 0) {
146             logger.debug("Previous value is greater than current value for metric '" + currentResult.getName() + "', ignore it");
147 
148             return new QueryResult(currentResult.getName(), "gauge", currentResult.getValue(), currentResult.getEpoch(MILLISECONDS));
149         }
150 
151         BigDecimal valueDelta;
152         if (currentValue instanceof Long) {
153             valueDelta = new BigDecimal(currentValue.longValue() - previousValue.longValue());
154         } else if (currentValue instanceof Integer) {
155             valueDelta = new BigDecimal(currentValue.intValue() - previousValue.intValue());
156         } else if (currentValue instanceof Float) {
157             valueDelta = new BigDecimal(currentValue.floatValue() - previousValue.floatValue());
158         } else if (currentValue instanceof Double) {
159             valueDelta = new BigDecimal(currentValue.doubleValue() - previousValue.doubleValue());
160         } else {
161             logger.info("unsupported value type '" + currentValue.getClass() + ", cannot calculate perMinute " + currentResult);
162 
163             return currentResult;
164         }
165 
166         // multiply by 1000 because duration will be in millis
167         // multiply by 60 for per-minute
168         BigDecimal perMinute = valueDelta.movePointRight(3).multiply(new BigDecimal(60)).divide(durationInMillis, RoundingMode.HALF_UP);
169 
170         Number newCurrentValue;
171         if (currentValue instanceof Long) {
172             newCurrentValue = perMinute.longValue();
173         } else if (currentValue instanceof Integer) {
174             newCurrentValue = perMinute.intValue();
175         } else if (currentValue instanceof Float) {
176             newCurrentValue = perMinute.floatValue();
177         } else if (currentValue instanceof Double) {
178             newCurrentValue = perMinute.doubleValue();
179         } else {
180             logger.info("Illegal state " + previousResult);
181 
182             return currentResult;
183         }
184 
185         return new QueryResult(currentResult.getName(), "gauge", newCurrentValue, currentResult.getEpoch(MILLISECONDS));
186     }

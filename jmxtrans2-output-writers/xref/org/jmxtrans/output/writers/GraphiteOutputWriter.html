23  package org.jmxtrans.output.writers;
24  
25  import java.io.IOException;
26  import java.io.Writer;
27  import java.net.InetAddress;
28  import java.net.InetSocketAddress;
29  import java.net.UnknownHostException;
30  import java.nio.charset.Charset;
31  import java.util.Map;
32  
33  import javax.annotation.Nonnull;
34  import javax.annotation.Nullable;
35  import javax.annotation.concurrent.ThreadSafe;
36  
37  import org.jmxtrans.core.output.OutputWriterFactory;
38  import org.jmxtrans.core.results.QueryResult;
39  import org.jmxtrans.output.writers.support.BatchingOutputWriter;
40  import org.jmxtrans.output.writers.support.TcpOutputWriter;
41  import org.jmxtrans.output.writers.support.WriterBasedOutputWriter;
42  import org.jmxtrans.utils.VisibleForTesting;
43  
44  import static java.util.concurrent.TimeUnit.SECONDS;
45  
46  import static org.jmxtrans.utils.ConfigurationUtils.getInt;
47  
48  @ThreadSafe
49  public class GraphiteOutputWriter implements WriterBasedOutputWriter {
50  
51      @Nullable private volatile String metricPathPrefix;
52  
53      @VisibleForTesting
54      GraphiteOutputWriter() {}
55  
56      @Override
57      public void write(@Nonnull Writer writer, @Nonnull QueryResult result) throws IOException {
58          writer.write(
59                  buildMetricPathPrefix()
60                          + result.getName() + " "
61                          + result.getValue() + " "
62                          + result.getEpoch(SECONDS));
63      }
64  
65      @Nonnull
66      private String buildMetricPathPrefix() {
67          // {@link java.net.InetAddress#getLocalHost()} may not be known at JVM startup when the process is launched as a Linux service.
68          // FIXME: there is a 5 second cache on localhost name, it probably make sense to reload it periodically. Hostname can change.
69          if (metricPathPrefix != null) return metricPathPrefix;
70  
71          metricPathPrefix = "servers." + getHostname() + ".";
72          return metricPathPrefix;
73      }
74  
75      @Nonnull
76      private String getHostname() {
77          try {
78              return InetAddress.getLocalHost().getHostName().replaceAll("\\.", "_");
79          } catch (UnknownHostException e) {
80              return  "#unknown#";
81          }
82      }
83  
84      public static class Factory implements OutputWriterFactory<BatchingOutputWriter<TcpOutputWriter<GraphiteOutputWriter>>> {
85          @Nonnull
86          @Override
87          public BatchingOutputWriter<TcpOutputWriter<GraphiteOutputWriter>> create(@Nonnull Map<String, String> settings) {
88  
89              String hostname = settings.get("hostname");
90              int port = getInt(settings, "port");
91              int socketTimeoutMillis = getInt(settings, "socketTimeoutMillis", 2000);
92              int batchSize = getInt(settings, "batchSite", 100);
93  
94              Charset charset = Charset.forName("UTF-8");
95              InetSocketAddress server = new InetSocketAddress(hostname, port);
96  
97              return new BatchingOutputWriter(
98                      batchSize,
99                      new TcpOutputWriter<>(
100                             server,
101                             socketTimeoutMillis,
102                             charset,
103                             new GraphiteOutputWriter())
104             );
105         }
106     }

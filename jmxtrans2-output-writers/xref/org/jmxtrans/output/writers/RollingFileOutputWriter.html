45  public class RollingFileOutputWriter extends AbstractOutputWriter {
46  
47      private static final Logger LOGGER = Logger.getLogger(IoUtils.class.getName());
48  
49      public final static String SETTING_FILE_NAME = "fileName";
50      public final static String SETTING_FILE_NAME_DEFAULT_VALUE = "jmxtrans-agent.data";
51      public final static String SETTING_MAX_FILE_SIZE = "maxFileSize";
52      public final static long SETTING_MAX_FILE_SIZE_DEFAULT_VALUE=10;
53      public final static String SETTING_MAX_BACKUP_INDEX = "maxBackupIndex";
54      public final static int SETTING_MAX_BACKUP_INDEX_DEFAULT_VALUE = 5; 
55      private static DateFormat dfISO8601 = new SimpleDateFormat("yyyy-MM-dd'T'HH:mm'Z'");
56      
57      protected Writer temporaryFileWriter;
58      protected File temporaryFile;
59      protected File file = new File(SETTING_FILE_NAME_DEFAULT_VALUE);
60      protected long maxFileSize;
61      protected int maxBackupIndex;
62  
63      public static void appendToFile(File source, File destination, long maxFileSize, int maxBackupIndex) throws IOException {
64          boolean destinationExists = validateDestinationFile(source, destination, maxFileSize, maxBackupIndex);
65          if (destinationExists) {
66              IoUtils.doCopySmallFile(source, destination, true);
67          } else {
68              boolean renamed = source.renameTo(destination);
69              if (!renamed) {
70                  IoUtils.doCopySmallFile(source, destination, false);
71              }
72          }
73      }
74  
75      // visible for testing
76      static boolean validateDestinationFile(File source, File destination, long maxFileSize, int maxBackupIndex) throws IOException {
77          if (!destination.exists() || destination.isDirectory()) return false;
78          long totalLengthAfterAppending = destination.length() + source.length();
79          if (totalLengthAfterAppending > maxFileSize) {
80              rollFiles(destination, maxBackupIndex);
81              return false; // File no longer exists because it was move to filename.1
82          }
83  
84          return true;
85      }
86  
87      // visible for testing
88      static void rollFiles(File destination, int maxBackupIndex) throws IOException {
89  
90          // if maxBackup index == 10 then we will have file
91          // outputFile, outpuFile.1 outputFile.2 ... outputFile.10
92          // we only care if 9 and lower exists to move them up a number
93          for (int i = maxBackupIndex - 1; i >= 0; i--) {
94              String path = destination.getAbsolutePath();
95              path=(i==0)?path:path + "." + i;
96              File f = new File(path);
97              if (!f.exists()) continue;
98  
99              File fNext = new File(destination + "." + (i + 1));
100             IoUtils.doCopySmallFile(f, fNext, false);
101         }
102 
103         if (!destination.delete()) {
104             LOGGER.log(Level.WARNING, "Could not delete file [" + destination.getAbsolutePath() + "].");
105         }
106     }
107 
108     /**
109      * Simple implementation without chunking if the source file is big.
110      *
111      * @param source
112      * @param destination
113      * @throws java.io.IOException
114      */
115     // visible for testing
116     public static void doCopySmallFile(@Nonnull File source, @Nonnull File destination) throws IOException {
117         if (destination.exists() && destination.isDirectory()) {
118             throw new IOException("Can not copy file, destination is a directory: " + destination.getAbsolutePath());
119         }
120 
121         FileInputStream fis = null;
122         FileOutputStream fos = null;
123         FileChannel input = null;
124         FileChannel output = null;
125         try {
126             fis = new FileInputStream(source);
127             fos = new FileOutputStream(destination, false);
128             input = fis.getChannel();
129             output = fos.getChannel();
130             output.transferFrom(input, 0, input.size());
131         } finally {
132             IoUtils.closeQuietly(output);
133             IoUtils.closeQuietly(input);
134             IoUtils.closeQuietly(fis);
135             IoUtils.closeQuietly(fos);
136         }
137 
138         if (destination.length() != source.length()) {
139             throw new IOException("Failed to copy content from '" +
140                     source + "' (" + source.length() + "bytes) to '" + destination + "' (" + destination.length() + ")");
141         }
142     }
143 
144     public static void replaceFile(File source, File destination) throws IOException {
145         if (destination.exists()) {
146             // try to delete destination, it might fail, but doCopySmallFile() can deal with it
147             if (!destination.delete()) {
148                 LOGGER.info("Could not delete " + destination.getAbsolutePath() + ", replacing only the content");
149             }
150         }
151         doCopySmallFile(source, destination);
152     }

154     @Override
155     public synchronized void postConstruct(Map<String, String> settings) {
156         super.postConstruct(settings);
157         TimeZone tz = TimeZone.getTimeZone("UTC");
158         dfISO8601.setTimeZone(tz);
159         file = new File(getString(settings, SETTING_FILE_NAME, SETTING_FILE_NAME_DEFAULT_VALUE));
160         maxFileSize = getLong(settings, SETTING_MAX_FILE_SIZE, SETTING_MAX_FILE_SIZE_DEFAULT_VALUE);
161         maxBackupIndex = getInt(settings, SETTING_MAX_BACKUP_INDEX, SETTING_MAX_BACKUP_INDEX_DEFAULT_VALUE);
162         // Performance related after 10mb the Filechanel output.transferTo() performs badly
163         if (maxFileSize > 10 || maxFileSize < 0) maxFileSize = 10;
164         maxFileSize = maxFileSize * 1000000; //converts to bytes.
165         logger.log(getInfoLevel(), "RollingFileOutputWriter configured with file " + file.getAbsolutePath());
166     }
167 
168     protected Writer getTemporaryFileWriter() throws IOException {
169         if (temporaryFile == null) {
170             temporaryFile = File.createTempFile("jmxtrans-agent-", ".data");
171             temporaryFile.deleteOnExit();
172             if (logger.isLoggable(getDebugLevel()))
173                 logger.log(getDebugLevel(), "Created temporary file " + temporaryFile.getAbsolutePath());
174 
175             temporaryFileWriter = null;
176         }
177         if (temporaryFileWriter == null) {
178             temporaryFileWriter = new BufferedWriter(new OutputStreamWriter(new FileOutputStream(temporaryFile, false), Charset.forName("UTF-8")));
179         }
180 
181         return temporaryFileWriter;
182     }
183 
184     public synchronized void writeQueryResult(@Nonnull String name, @Nullable String type, @Nullable Object value) throws IOException {
185         try {
186             getTemporaryFileWriter().write("["+dfISO8601.format(Calendar.getInstance().getTime()) +"] "+name + " " + value + "\n");
187         } catch (IOException e) {
188             releaseTemporaryWriter();
189             throw e;
190         }
191     }
192 
193     @Override
194     public synchronized void write(QueryResult result) throws IOException {
195         try {
196             getTemporaryFileWriter().write("["+dfISO8601.format(Calendar.getInstance().getTime()) +"] "+ result.getName() + " " + result.getValue() + "\n");
197         } catch (IOException e) {
198             releaseTemporaryWriter();
199             throw e;
200         }
201     }
202 
203     protected void releaseTemporaryWriter() {
204         try {
205             IoUtils.closeQuietly(getTemporaryFileWriter());
206         } catch (IOException ignore) {
207         }
208         if (temporaryFile != null) {
209             if (!temporaryFile.delete()) {
210                 logger.log(Level.WARNING, "Could not delete temporary file [" + temporaryFile.getAbsolutePath() + "].");
211             }
212         }
213         temporaryFile = null;
214 
215     }
216 
217     @Override
218     public synchronized void postCollect() throws IOException {
219         try {
220             getTemporaryFileWriter().close();
221             if (logger.isLoggable(getDebugLevel()))
222                 logger.log(getDebugLevel(), "Overwrite " + file.getAbsolutePath() + " by " + temporaryFile.getAbsolutePath());
223             appendToFile(temporaryFile, file, maxFileSize, maxBackupIndex);
224         } finally {
225             temporaryFileWriter = null;
226         }
227     }

65      @Test
66      public void basic_jmx_attribute_return_simple_result() throws Exception {
67          EmbeddedJmxTrans embeddedJmxTrans = new EmbeddedJmxTrans();
68  
69          Query query = new Query("test:type=MemoryPool,name=PS Eden Space").addAttribute("CollectionUsageThreshold");
70          embeddedJmxTrans.addQuery(query);

72          assertThat(query.getResults().size(), is(1));

95      @Test
96      public void testExportResults() {
97          EmbeddedJmxTrans embeddedJmxTrans = new EmbeddedJmxTrans();
98  
99          // CONFIGURE
100         Query query = new Query("test:type=GarbageCollector,name=PS Scavenge");
101         embeddedJmxTrans.addQuery(query);
102         query.addAttribute("CollectionCount").addAttribute("CollectionTime");
103         embeddedJmxTrans.addQuery(query);
104 
105         final AtomicInteger exportCount = new AtomicInteger();
106         final AtomicInteger exportResultCount = new AtomicInteger();
107 
108         OutputWriter outputWriter = new AbstractOutputWriter() {
109             @Override
110             public void write(Iterable<QueryResult> results) {
111                 exportCount.incrementAndGet();
112                 for (QueryResult result : results) {
113                     exportResultCount.incrementAndGet();
114                 }
115             }
116         };
117 
118         embeddedJmxTrans.getOutputWriters().add(outputWriter);
119         assertThat(query.getOutputWriters().size(), is(0));
120         assertThat(query.getEffectiveOutputWriters().size(), is(1));
121 
122         // PREPARE DATA
123         long time = System.currentTimeMillis() - TimeUnit.MILLISECONDS.convert(5, TimeUnit.MINUTES);
124         for (int i = 0; i < 100; i++) {
125             QueryResult result = new QueryResult("PS_Scavenge.GarbageCollector.CollectionTime", 5 * i, time);
126             query.getResults().add(result);
127 
128             assertThat(query.getResults().size(), is(i + 1));
129 
130 
131             time += TimeUnit.MILLISECONDS.convert(10, TimeUnit.SECONDS);
132         }
133 
134         // TEST
135         int actualExportResultCount = query.exportCollectedMetrics();
136         assertThat(exportCount.get(), is(2));
137         assertThat(exportResultCount.get(), is(100));
138         assertThat(actualExportResultCount, is(100));
139     }

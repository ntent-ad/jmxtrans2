23  package org.jmxtrans.embedded.output;
24  
25  import org.jmxtrans.embedded.QueryResult;
26  import org.slf4j.Logger;
27  import org.slf4j.LoggerFactory;
28  
29  import java.util.concurrent.TimeUnit;
30  
31  /**
32   * <a href="http://www.slf4j.org/">SLF4J</a> based {@linkplain OutputWriter} implementation.
33   * <p/>
34   * Settings:
35   * <ul>
36   * <li>"logger": Name of the logger. Optional, default value: "<code>org.jmxtrans.embedded.output.Slf4jWriter</code>"</li>
37   * <li>"enabled": flag to enable/disable the writer. Optional, default value: <code>true</code>.</li>
38   * </ul>
39   * <p/>
40   * Output: Graphite's <a href="http://graphite.readthedocs.org/en/0.9.10/feeding-carbon.html#the-plaintext-protocol">
41   * Carbon Plan Text protocol</a>
42   * <pre>
43   *     <code>&lt;metric path&gt; &lt;metric value&gt; &lt;metric timestamp&gt;.</code>
44   * </pre>
45   * With timestamp in seconds.
46   *
47   * @author <a href="mailto:cleclerc@xebia.fr">Cyrille Le Clerc</a>
48   */
49  public class Slf4jWriter extends AbstractOutputWriter {
50  
51      private Logger logger = LoggerFactory.getLogger(getClass());
52  
53      /**
54       * Initialize the logger.
55       */
56      @Override
57      public void start() {
58          super.start();
59          logger = LoggerFactory.getLogger(getStringSetting("logger", getClass().getName()));
60      }
61  
62      @Override
63      public void write(Iterable<QueryResult> results) {
64          for (QueryResult result : results) {
65              String msg = result.getName() + " " + result.getValue() + " " + result.getEpoch(TimeUnit.SECONDS);
66              logger.info(msg);
67          }
68      }
69  }

43  public class LibratoWriterTest {
44  
45      private LibratoWriter libratoWriter;
46      private ByteArrayOutputStream out;
47      private QueryResult counterResult1;
48      private QueryResult counterResult2;
49      private QueryResult gaugeResult1;
50      private QueryResult gaugeResult2;
51      private QueryResult unkownType;
52      private QueryResult noType;
53  
54      private final ObjectMapper objectMapper = new ObjectMapper();
55  
56      @BeforeClass
57      public void createLibratoWriter() {
58          libratoWriter = new LibratoWriter(new JsonFactory(), "myHost.test.net");
59      }
60  
61      @BeforeMethod
62      public void createOutputStream() {
63          out = new ByteArrayOutputStream();
64      }
65  
66      @BeforeClass
67      public void createResults() {
68          counterResult1 = new QueryResult("counter1", "counter", 1, 1000L);
69          counterResult2 = new QueryResult("counter2", "counter", 2L, 2000L);
70          gaugeResult1 = new QueryResult("gauge1", "gauge", 3.3, 3000L);
71          gaugeResult2 = new QueryResult("gauge2", "gauge", 4.4f, 4000L);
72          unkownType = new QueryResult("unkownType", "unkownType", 5, 5000L);
73          noType = new QueryResult("noType", null, 6, 6000L);
74      }
75  
76      @Test
77      public void canWriteSimpleResults() throws IOException {
78          libratoWriter.beforeBatch(out);
79          libratoWriter.write(out, counterResult1);
80          libratoWriter.write(out, counterResult2);
81          libratoWriter.write(out, gaugeResult1);
82          libratoWriter.write(out, gaugeResult2);
83          libratoWriter.afterBatch(out);
84  
85          JsonNode tree = objectMapper.readTree(out.toByteArray());
86          JsonNode counters = tree.findPath("counters");
87          assertThat(counters.isArray()).isTrue();
88          assertThat(counters.size()).isEqualTo(2);
89  
90          JsonNode counter1 = counters.get(0);
91          assertThat(counter1.get("name").textValue()).isEqualTo("counter1");
92          assertThat(counter1.get("source").textValue()).isEqualTo("myHost.test.net");
93          assertThat(counter1.get("measure_time").intValue()).isEqualTo(1);
94          assertThat(counter1.get("value").intValue()).isEqualTo(1);
95  
96          JsonNode counter2 = counters.get(1);
97          assertThat(counter2.get("name").textValue()).isEqualTo("counter2");
98          assertThat(counter2.get("source").textValue()).isEqualTo("myHost.test.net");
99          assertThat(counter2.get("measure_time").intValue()).isEqualTo(2);
100         assertThat(counter2.get("value").longValue()).isEqualTo(2L);
101 
102         JsonNode gauges = tree.findPath("gauges");
103         assertThat(gauges.isArray()).isTrue();
104 
105         JsonNode gauge1 = gauges.get(0);
106         assertThat(gauge1.get("name").textValue()).isEqualTo("gauge1");
107         assertThat(gauge1.get("source").textValue()).isEqualTo("myHost.test.net");
108         assertThat(gauge1.get("measure_time").intValue()).isEqualTo(3);
109         assertThat(gauge1.get("value").doubleValue()).isEqualTo(3.3);
110 
111         JsonNode gauge2 = gauges.get(1);
112         assertThat(gauge2.get("name").textValue()).isEqualTo("gauge2");
113         assertThat(gauge2.get("source").textValue()).isEqualTo("myHost.test.net");
114         assertThat(gauge2.get("measure_time").intValue()).isEqualTo(4);
115         assertThat(gauge2.get("value").floatValue()).isEqualTo(4.4f);
116     }
117 
118     @Test
119     public void resultCountIsCorrect() throws IOException {
120         int counter = 0;
121         libratoWriter.beforeBatch(out);
122         counter += libratoWriter.write(out, counterResult1);
123         counter += libratoWriter.write(out, gaugeResult1);
124         counter += libratoWriter.afterBatch(out);
125 
126         assertThat(counter).isEqualTo(2);
127     }

129     @Test
130     public void nonNumericValuesAreSentEmpty() throws IOException {
131         libratoWriter.beforeBatch(out);
132         libratoWriter.write(out, new QueryResult("nonNumeric", "counter", "stringValue", 1000));
133         libratoWriter.afterBatch(out);
134 
135         JsonNode tree = objectMapper.readTree(out.toByteArray());
136         JsonNode counters = tree.findPath("counters");
137         assertThat(counters.isArray()).isTrue();
138         assertThat(counters.size()).isEqualTo(1);
139 
140         JsonNode counter1 = counters.get(0);
141         assertThat(counter1.get("name").textValue()).isEqualTo("nonNumeric");
142         assertThat(counter1.get("value")).isNull();
143     }
